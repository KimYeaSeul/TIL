switch-case 문법

if로 구현

var score = prompt('점수를 입력하세요');
if (score === 10) {
console.log('A');
} else if (score == 9) {
console.log('B');
} else {
console.log('C');
}

switch-case 로 구현

var score = prompt('점수를 입력하세요');
switch (score) {
case 10:
console.log('A);
break;
case 9:
console.log('B');
break;
default:
console.log('C');
}
